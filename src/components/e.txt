# Axios API Setup

## src/api/api.js

```jsx
import axios from "axios";

const API = axios.create({
  baseURL: "http://localhost:5000/api",
});

API.interceptors.request.use((req) => {
  const token = localStorage.getItem("token");

  if (token) {
    req.headers.Authorization = `Bearer ${token}`;
  }

  return req;
});

export default API;
```

## Login.jsx

```jsx
import React, { useState } from "react";
import { Link, useNavigate } from "react-router-dom";
import API from "../api/api";
import toast from "react-hot-toast";

const Login = () => {
  const navigate = useNavigate();

  const [formData, setFormData] = useState({
    email: "",
    password: "",
  });

  const handleLogin = async (e) => {
    e.preventDefault();

    try {
      const response = await API.post("/auth/login", formData);

      localStorage.setItem("token", response.data.token);

      toast.success("Login successful");

      navigate("/dashboard");
    } catch (err) {
      toast.error("Invalid credentials");
    }
  };

  return (
    <div className="min-h-screen bg-[#0f172a] flex items-center justify-center px-4">
      <div className="w-full max-w-md bg-[#1e293b] rounded-3xl p-8 shadow-2xl">
        <h1 className="text-3xl font-bold text-white text-center mb-2">
          Welcome Back
        </h1>

        <p className="text-gray-400 text-center mb-8">
          Login to continue your interview journey
        </p>

        <form onSubmit={handleLogin} className="space-y-5">
          <input
            type="email"
            placeholder="Email"
            className="w-full p-4 rounded-xl bg-[#0f172a] text-white border border-gray-700 outline-none"
            onChange={(e) =>
              setFormData({ ...formData, email: e.target.value })
            }
          />

          <input
            type="password"
            placeholder="Password"
            className="w-full p-4 rounded-xl bg-[#0f172a] text-white border border-gray-700 outline-none"
            onChange={(e) =>
              setFormData({ ...formData, password: e.target.value })
            }
          />

          <button className="w-full bg-blue-500 hover:bg-blue-600 transition-all text-white py-4 rounded-xl font-semibold">
            Login
          </button>
        </form>

        <p className="text-gray-400 text-center mt-6">
          Don't have an account?{' '}
          <Link to="/register" className="text-blue-400">
            Register
          </Link>
        </p>
      </div>
    </div>
  );
};

export default Login;
```

```jsx
import React, { useState } from "react";
import API from "../api/api";
import toast from "react-hot-toast";
import { useNavigate } from "react-router-dom";

const GenerateInterview = () => {
  const navigate = useNavigate();

  const [loading, setLoading] = useState(false);

  const [formData, setFormData] = useState({
    role: "",

          Generate AI Interview
        </h1>

        <p className="text-gray-400 mb-8">
          Create personalized interview questions instantly.
        </p>

        <form onSubmit={handleGenerate} className="space-y-6">
          <div>
            <label className="text-gray-300 block mb-2">
              Job Role
            </label>

            <input
              type="text"
              placeholder="MERN Stack Developer"
              className="w-full p-4 rounded-xl bg-[#0f172a] border border-gray-700 text-white outline-none"
              onChange={(e) =>
                setFormData({ ...formData, role: e.target.value })
              }
            />
          </div>

            <label className="text-gray-300 block mb-2">
              Experience Level
            </label>

            <select
              className="w-full p-4 rounded-xl bg-[#0f172a] border border-gray-700 text-white outline-none"
              onChange={(e) =>
                setFormData({ ...formData, level: e.target.value })
              }

              <option>Fresher</option>
              <option>Intermediate</option>

          </div>
```

# Interview Session Page

## InterviewSession.jsx

```jsx
import React, { useEffect, useState } from "react";
import { useParams, useNavigate } from "react-router-dom";
import API from "../api/api";
import toast from "react-hot-toast";

const InterviewSession = () => {
  const { id } = useParams();
  const navigate = useNavigate();

  const [questions, setQuestions] = useState([]);
  const [currentQuestion, setCurrentQuestion] = useState(0);
  const [answers, setAnswers] = useState([]);
  const [answer, setAnswer] = useState("");
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetchInterview();
  }, []);

  const fetchInterview = async () => {
    try {
      const response = await API.get(`/interview/${id}`);

      setQuestions(response.data.questions);
    } catch (err) {
      toast.error("Failed to load interview");
    } finally {
      setLoading(false);
    }
  };

  const handleNext = () => {
    const updatedAnswers = [...answers, answer];

    setAnswers(updatedAnswers);
    setAnswer("");

    if (currentQuestion + 1 < questions.length) {
      setCurrentQuestion(currentQuestion + 1);
    } else {
      submitInterview(updatedAnswers);
    }
  };

  const submitInterview = async (finalAnswers) => {
    try {
      const response = await API.post(`/interview/submit/${id}`, {
        answers: finalAnswers,
      });

      toast.success("Interview submitted successfully");

      navigate(`/result/${response.data.resultId}`);
    } catch (err) {
      toast.error("Failed to submit interview");
    }
  };

  if (loading) {
    return (
      <div className="min-h-screen bg-[#0f172a] flex items-center justify-center text-white text-2xl">
        Loading...
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-[#0f172a] text-white px-4 py-10">
      <div className="max-w-4xl mx-auto bg-[#1e293b] rounded-3xl p-8 shadow-2xl">
        <div className="flex items-center justify-between mb-8">
          <h1 className="text-3xl font-bold">
            AI Mock Interview
          </h1>

          <span className="bg-blue-500 px-4 py-2 rounded-lg">
            Question {currentQuestion + 1}/{questions.length}
          </span>
        </div>

        <div className="bg-[#0f172a] p-6 rounded-2xl mb-6">
          <h2 className="text-2xl font-semibold leading-relaxed">
            {questions[currentQuestion]}
          </h2>
        </div>

        <textarea
          value={answer}
          onChange={(e) => setAnswer(e.target.value)}
          placeholder="Write your answer here..."
          className="w-full h-52 bg-[#0f172a] border border-gray-700 rounded-2xl p-5 outline-none resize-none"
        />

        <button
          onClick={handleNext}
          className="w-full mt-6 bg-blue-500 hover:bg-blue-600 transition-all py-4 rounded-xl font-semibold"
        >
          {currentQuestion + 1 === questions.length
            ? "Submit Interview"
            : "Next Question"}
        </button>
      </div>
    </div>
  );
};

export default InterviewSession;
```

# Result Page

```jsx
import React, { useEffect, useState } from "react";
import { useParams } from "react-router-dom";
import API from "../api/api";
import toast from "react-hot-toast";

const Result = () => {
  const { id } = useParams();

  const [result, setResult] = useState(null);

  useEffect(() => {
    fetchResult();
  }, []);

  const fetchResult = async () => {

      const response = await API.get(`/result/${id}`);

      setResult(response.data);

      toast.error("Failed to fetch result");
```

```jsx
import React, { useEffect, useState } from "react";
import API from "../api/api";
import toast from "react-hot-toast";

const History = () => {
  const [history, setHistory] = useState([]);

  useEffect(() => {
    fetchHistory();
  }, []);

  const fetchHistory = async () => {
    try {
      const response = await API.get("/interview/history");

      setHistory(response.data);
    } catch (err) {
      toast.error("Failed to load history");
    }
  };

  return (
    <div className="min-h-screen bg-[#0f172a] text-white px-4 py-10">
      <div className="max-w-6xl mx-auto">
        <div className="flex items-center justify-between mb-10">
          <div>
            <h1 className="text-4xl font-bold">
              Interview History
            </h1>

            <p className="text-gray-400 mt-2">
              Track all your previous AI interviews
            </p>
          </div>
        </div>

        <div className="space-y-5">
          {history.map((item) => (
            <div
              key={item._id}
              className="bg-[#1e293b] rounded-2xl p-6 flex items-center justify-between"
            >
              <div>
                <h2 className="text-2xl font-semibold">
                  {item.role}
                </h2>

                <p className="text-gray-400 mt-1">
                  {item.level} Level Interview
                </p>
              </div>

              <div className="text-right">
                <div className="bg-blue-500 px-5 py-3 rounded-xl text-xl font-bold inline-block">
                  {item.score}%
                </div>

                <p className="text-gray-400 mt-2 text-sm">
                  {new Date(item.createdAt).toLocaleDateString()}
                </p>
              </div>
            </div>
          ))}
        </div>
      </div>
    </div>
  );
};

export default History;
```
